void ActPhysics::SilMatrix::MoveXYTo ( double  xRef,
const std::pair< double, double > &  yzCentre,
double  xTarget 
)
145{
146 // Silicon matrix is obtained at a given distance from the origin
147 // This function allows scalling and moving that distance
148 // by simply using triangle equivalence
149 auto scale {xTarget / xRef};
150 for(auto& [idx, g] : fMatrix)
151 {
152 for(int i = 0, size = g->GetN(); i < size; i++)
153 {
154 auto y {g->GetPointX(i)};
155 auto z {g->GetPointY(i)};
156 // But reference is not the origin but the given pair!
157 auto diffy {y - yzCentre.first};
158 auto diffz {z - yzCentre.second};
159 // And we scale THAT DIFFERENCE
160 auto scaledy {diffy * scale};
161 auto scaledz {diffz * scale};
162 // And set new points
163 // Y = Y + (scaledDiffY - diffY);
164 g->SetPointX(i, y + (scaledy - diffy));
165 g->SetPointY(i, z + (scaledz - diffz));
166 }
167 }
168}

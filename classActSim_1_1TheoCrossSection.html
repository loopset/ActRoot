Detailed Description

A class to implement theorethical cross sections in simulations.

To be used in the same fashion as Geant4 does Computes the xs dotting the absolute * angular distributions using Legendre coefficients

130 int nLegendre {std::stoi(nLegendreString)};
131
132 //SWITCH ACCORDING TO NUM OF NLEGENDRE
133 int linesToParse {};
134 bool stop {false};
135 switch (nLegendre)
136 {
137 case 1 ... 6:
138 //just parse one line
139 linesToParse = 1;
140 break;
141 case 7 ... 12:
142 linesToParse = 2;
143 break;
144 case 13 ... 18:
145 linesToParse = 3;
146 break;
147 default:
148 stop = true;
149 break;
150 }
151 //reached end of data points
152 if(stop)
153 break;
154 //parse lines
155 std::vector<double> coeffs {};
156 for(int i = 0; i < linesToParse; i++)
157 {
158 std::getline(streamer, line);
159 ParseLine(line, coeffs);
160 }
161 //and write coeffs to map
162 for(int i = 1; i <= nLegendre; i++)
163 {
164 mAngular[i].first.push_back(energy * converteVToMeV);
165 mAngular[i].second.push_back(coeffs[i - 1]);
166 }
167 }
168 streamer.close();

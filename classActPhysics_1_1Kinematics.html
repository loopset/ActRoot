TGraphErrors * ActPhysics::Kinematics::TransfromCMCrossSectionToLab ( TGraphErrors *  gcm)

647{
648 // Get CM vs Lab line
649 auto* gangles {GetThetaLabvsThetaCMLine()};
650 // Set ranges for
651 double labmax {180}; // deg
652 double cmmax {180};
653 // But if we are in inverse kinematics, that graph is bivaluated in the lab frame
654 // and therefore we must cut in the maximum to pick only one region. Default is [0, max]
655 // But this could depend on the reaction. This is adapted for (p,d) or (d,t) reactions
656 if(fInverse)
657 {
658 // Locate maximum
659 auto it {TMath::LocMax(gangles->GetN(), gangles->GetY())};
660 // And set values, with a safe radius to avoid divergencies in the xs (mainly due to the derivative)!
661 labmax = gangles->GetPointY(it) - 1;
662 cmmax = gangles->GetPointX(it) - 1;
663 }
664
665 // Compute jacobian of transformation CM / Lab
666 auto gder {std::make_unique<TGraphErrors>()};
667 for(int p = 0; p < gangles->GetN(); p++)
668 {
669 auto cm {gangles->GetPointX(p)};
670 auto lab {gangles->GetPointY(p)};
671 if(cm < cmmax)
672 {
673 // X axis = cos(lab)
674 // Y axis = cos(cm)
675 gder->AddPoint(TMath::Cos(lab * TMath::DegToRad()), TMath::Cos(cm * TMath::DegToRad()));
676 }
677 }
678 // And declare function!
679 auto fder {
680 std::make_unique<TF1>("fder", [&](double* x, double* p) { return gder->Eval(x[0], nullptr, "S"); }, -1, 1, 0)};
681 // And transform cross-section!
682 // Using same binning as theoretical xs
683 auto* glab {new TGraphErrors};
684 glab->SetTitle(";#theta_{Lab} [#circ];d#sigma / d#Omega [mb/sr]");
685 double step {TMath::Abs(gcm->GetPointX(1) - gcm->GetPointX(0))};
686 for(double lab = 0; lab <= labmax; lab += step)
687 {
688 auto coslab {TMath::Cos(lab * TMath::DegToRad())};
689 auto coscm {gder->Eval(coslab)};
690 auto cm {TMath::ACos(coscm) *
691 TMath::RadToDeg()}; // too complicated but done this way to avoid creating another graph
692 auto jacobian {fder->Derivative(coslab)};
693 auto xscm {gcm->Eval(cm, nullptr, "S")};
694 glab->AddPoint(lab, xscm * jacobian);
695 }

212{
213 // Usually a sampled reaction returns the angles in the beam frame
214 // But to compute geometrical things (propagate that track to a silicon)
215 // We need to work in the "geometry = world" frame,
216 // where in ACTAR the "beam" goes in {1, 0, 0}
217 // Using XYZPoint and XYZVector is easy to compute it,
218 // as quoted here: https://root-forum.cern.ch/t/get-3x3-rotation-matrix-between-two-tvector3/60070
219 // Using only GenVector classes
220 auto originalFrame {beamDir.Unit()};
221 XYZVector worldFrame {1, 0, 0};
222 auto cross {worldFrame.Cross(originalFrame)}; // this defines the rotation axis
223 auto angle {TMath::ACos(originalFrame.Dot(worldFrame))};
224 ROOT::Math::AxisAngle axis {cross, angle};
225 ROOT::Math::Rotation3D rotation {axis};
226 return rotation(vBeamFrame);
227}
